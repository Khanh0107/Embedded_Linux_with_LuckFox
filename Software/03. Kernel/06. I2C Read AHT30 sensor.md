# Read AHT30 sensor

This guide explains how to implement and use a Linux kernel driver for the AHT30 temperature and humidity sensor (ASAIR, I2C address 0x38).


## Step 1: Device Tree Configuration

Add a new node for the potentiometer in your `.../luckfox-pico/sysdrv/source/kernel/arch/arm/boot/dts/rv1106g-luckfox-pico-ultra-w.dts` device tree file:

```dts
&i2c2 {
	status = "okay";
	aht30@38 {
		compatible = "asair,aht30";
		reg = <0x38>;		/* 7-bit I2C address */
	};
};
```

## Step 2: Implement AHT30 Reading Function in Kernel 

Create driver source file `aht30.c` in `../kernel/drivers/iio/humidity/aht30.c`

```c
// SPDX-License-Identifier: GPL-2.0
#include <linux/module.h>
#include <linux/i2c.h>
#include <linux/sysfs.h>
#include <linux/delay.h>
#include <linux/mutex.h>

#define AHT30_I2C_ADDR     0x38
#define AHT30_CMD_INIT     0xBE
#define AHT30_CMD_MEAS     0xAC
#define AHT30_STATUS_BUSY  BIT(7)
#define AHT30_MEAS_DELAY   80
#define AHT30_DATA_LEN     7

static struct i2c_client *client;
static DEFINE_MUTEX(aht30_lock);
static int aht30_temp_dc;   /* °C x 100  */
static int aht30_rh_pc;     /* %RH x 100 */

/* ---------- sysfs ---------- */
static int aht30_update(void)
{
	u8 cmd[3] = {AHT30_CMD_MEAS, 0x33, 0x00};
	u8 buf[AHT30_DATA_LEN];
	u32 raw_h, raw_t;
	int ret, tries;

	mutex_lock(&aht30_lock);

	/* Gửi lệnh đo */
	for (tries = 0; tries < 3; tries++) {
		ret = i2c_master_send(client, cmd, sizeof(cmd));
		if (ret == sizeof(cmd)) break;
		dev_err(&client->dev, "aht30: send meas cmd fail (%d), try %d\n", ret, tries + 1);
		msleep(20);
	}
	if (ret != sizeof(cmd)) {
		dev_err(&client->dev, "aht30: give up after 3 tries\n");
		goto err_out;
	}

	msleep(AHT30_MEAS_DELAY);

	/* Chờ ready */
	for (tries = 0; tries < 50; tries++) {
		ret = i2c_smbus_read_byte(client);
		if (ret < 0) {
			dev_err(&client->dev, "aht30: read status fail (%d)\n", ret);
			goto err_out;
		}
		if (!(ret & AHT30_STATUS_BUSY)) break;
		msleep(20);
	}
	if (tries == 50) {
		dev_err(&client->dev, "aht30: sensor still busy after 1 s\n");
		ret = -ETIMEDOUT;
		goto err_out;
	}

	/* ĐỌC DỮ LIỆU KHÔNG QUA THANH GHI */
	ret = i2c_master_recv(client, buf, AHT30_DATA_LEN);
	if (ret != AHT30_DATA_LEN) {
		dev_err(&client->dev, "aht30: read data fail (%d)\n", ret);
		goto err_out;
	}

	/* Kiểm tra status byte (buf[0]) nếu cần */
	raw_h = (buf[1] << 12) | (buf[2] << 4) | (buf[3] >> 4);
	raw_t = ((buf[3] & 0x0F) << 16) | (buf[4] << 8) | buf[5];

	aht30_temp_dc = (raw_t * 20000LL >> 20) - 5000;  /* °C x100 */
	aht30_rh_pc   = (raw_h * 10000LL >> 20);          /* %RH x100 */

	ret = 0;
	goto out;

err_out:
	mutex_unlock(&aht30_lock);
	return ret;
out:
	mutex_unlock(&aht30_lock);
	return ret;
}

static ssize_t temp1_show(struct device *dev, struct device_attribute *attr, char *buf)
{
	if (aht30_update()) return -EIO;
	return sprintf(buf, "%d.%02d\n", aht30_temp_dc / 100, abs(aht30_temp_dc) % 100);
}

static ssize_t humid1_show(struct device *dev, struct device_attribute *attr, char *buf)
{
	if (aht30_update()) return -EIO;
	return sprintf(buf, "%d.%02d\n", aht30_rh_pc / 100, aht30_rh_pc % 100);
}

static DEVICE_ATTR(temp1, 0444, temp1_show, NULL);
static DEVICE_ATTR(humid1, 0444, humid1_show, NULL);

static struct attribute *aht30_attrs[] = {
	&dev_attr_temp1.attr,
	&dev_attr_humid1.attr,
	NULL
};

static const struct attribute_group aht30_attr_group = {
	.attrs = aht30_attrs,
};

/* ---------- probe / remove ---------- */
static int aht30_probe(struct i2c_client *cli, const struct i2c_device_id *id)
{
	u8 init_cmd[3] = {0xBE, 0x08, 0x00}; // full init command
	int ret;

	if (!i2c_check_functionality(cli->adapter, I2C_FUNC_I2C))
		return -ENODEV;

	client = cli;

	ret = i2c_master_send(client, init_cmd, sizeof(init_cmd));
	if (ret != sizeof(init_cmd))
		return -EIO;
	msleep(40);

	ret = sysfs_create_group(&client->dev.kobj, &aht30_attr_group);
	if (ret)
		return ret;

	dev_info(&client->dev, "aht30: temp1 / humid1 sysfs ready\n");
	return 0;
}

static int aht30_remove(struct i2c_client *cli)
{
	sysfs_remove_group(&cli->dev.kobj, &aht30_attr_group);
	return 0;
}

static const struct of_device_id aht30_of_match[] = {
	{ .compatible = "asair,aht30" },
	{ /* sentinel */ }
};
MODULE_DEVICE_TABLE(of, aht30_of_match);

static struct i2c_driver aht30_driver = {
	.driver = {
		.name = "aht30",
		.of_match_table = aht30_of_match,
	},
	.probe    = aht30_probe,
	.remove   = aht30_remove,
};
module_i2c_driver(aht30_driver);

MODULE_AUTHOR("LuckFox User");
MODULE_DESCRIPTION("AHT30 – direct temp1 / humid1 sysfs, retry+log");
MODULE_LICENSE("GPL");

```

Explaination:

- Initializes the AHT30 sensor via I2C.
- Periodically sends measurement commands and reads temperature/humidity data.
- Provides two sysfs attributes (temp1, humid1) for user-space to read sensor values.
- Handles retries, error logging, and synchronization with a mutex.
- Registers the driver for device tree matching (asair,aht30) and cleans up on removal.

## Step 3: Create make file 

Add `obj-$(CONFIG_AHT30) += aht30.o` in `Makefile` file from `../source/kernel/drivers/iio/humidity/Makefile`

## Step 4 Add option AHT30 in kernelconfig

Add option AHT30 in `Kconfig` file from `.../source/kernel/drivers/iio/humidity/Kconfig`

```Kconfig
...
config AHT30
	tristate "AHT30 humidity and temperature sensor"
	depends on I2C
	select IIO_BUFFER
	help
	  Support for the AHT30 I2C humidity and temperature sensor.

	  To compile this driver as a module, choose M here: the module
	  will be called aht30.

endmenu
```

## Step 5: Create kernel config and build 

Run these commands to configure and build the kernel:

```sh
make ARCH=arm luckfox_rv1106_linux_defconfig    (config file in /configs folder)
make ARCH=arm menuconfig                        (show config UI, remember to save before exit)
make ARCH=arm savedefconfig                     (generate defconfig file)
mv defconfig arch/arm/configs/luckfox_rv1106_linux_defconfig  (move new created file back to /configs folder)
```

- Make sure `CONFIG_AHT30=y` in `luckfox_rv1106_linux_defconfig` file

- At `../luckfox-pico/`  and run command

```sh
./build.sh clean kernel
./build.sh kernel
```

- After flasing, test read humid and temperature by

```sh
[root@luckfox root]# cat ../sys/bus/i2c/drivers/aht30/2-0038/temp1
32.60
[root@luckfox root]# cat ../sys/bus/i2c/drivers/aht30/2-0038/humid1
68.26
```

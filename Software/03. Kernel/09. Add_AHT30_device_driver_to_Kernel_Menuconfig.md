# Adding AHT30 Sensor Driver to Kernel Menuconfig for Luckfox Pico Ultra W

This guide provides step-by-step instructions to add the AHT30 humidity and temperature sensor package to the kernel menuconfig using the Luckfox Pico Ultra W manufacturer's SDK.

## Prerequisites

- Luckfox Pico Ultra W SDK downloaded and set up
- Basic knowledge of Linux command line and kernel build process

---

## 1. Add the AHT30 Driver Source

1. Navigate to the driver directory:
    ```bash
    cd Build_Luckfox/luckfox-pico/sysdrv/source/kernel/drivers/iio/humidity
    ```
2. Create a new file for the AHT30 driver:
    ```bash
    touch aht30.c
    ```
3. Implement the driver code in `aht30.c` (refer to the manufacturer's documentation or existing sensor drivers for guidance).

Example:

```c
#include <linux/module.h>
#include <linux/init.h>
#include <linux/i2c.h>
#include <linux/delay.h>
#include <linux/slab.h>
#include <linux/of.h>
#include <linux/hwmon.h>
#include <linux/hwmon-sysfs.h>
#include <linux/err.h>
#include <linux/mutex.h>


#define AHT30_I2C_ADDR          0x38  // I2C address of AHT30 sensor
#define AHT30_INIT_CMD          0xBE  // Sensor initialization command
#define AHT30_MEASURE_CMD       0xAC  // Measurement command
#define AHT30_RESET_CMD         0xBA  // Sensor reset command

#define AHT30_STATUS_BUSY       0x80  // Status bit: sensor is busy
#define AHT30_STATUS_CALIBRATED 0x08  // Status bit: sensor is calibrated

#define AHT30_MEASURE_DELAY     80  /* Wait time after measurement command (ms) */
#define AHT30_RESET_DELAY       20  /* Wait time after reset (ms) */
#define AHT30_INIT_DELAY        10  /* Wait time after initialization (ms) */


// Structure to store AHT30 sensor data and state
struct aht30_data {
    struct i2c_client *client;      // I2C client object
    struct mutex lock;              // Mutex for data access protection
    bool valid;                     // Flag indicating valid data
    unsigned long last_updated;     // Last data update timestamp
    int temperature;                // Temperature 
    int humidity;                   // Humidity 
};


/*
 * Send command to AHT30 sensor via I2C
 * cmd: command code
 * data: additional data (if any)
 * len: length of additional data
 */
static int aht30_send_cmd(struct i2c_client *client, u8 cmd, u8 *data, int len)
{
    u8 buf[3];
    int ret;

    buf[0] = cmd;
    if (data && len > 0) {
        memcpy(&buf[1], data, len);
    }

    ret = i2c_master_send(client, buf, len + 1);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to send command 0x%02x: %d\n", cmd, ret);
        return ret;
    }

    return 0;
}


/*
 * Read data from AHT30 sensor via I2C
 * buf: buffer to store read data
 * len: number of bytes to read
 */
static int aht30_read_data(struct i2c_client *client, u8 *buf, int len)
{
    int ret;

    ret = i2c_master_recv(client, buf, len);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to read data: %d\n", ret);
        return ret;
    }

    return 0;
}


/*
 * Reset the AHT30 sensor
 */
static int aht30_reset(struct i2c_client *client)
{
    int ret;

    ret = aht30_send_cmd(client, AHT30_RESET_CMD, NULL, 0);
    if (ret < 0)
        return ret;

    msleep(AHT30_RESET_DELAY);
    return 0;
}


/*
 * Initialize the AHT30 sensor (send init command)
 */
static int aht30_init(struct i2c_client *client)
{
    u8 data[2] = {0x08, 0x00};
    int ret;

    ret = aht30_send_cmd(client, AHT30_INIT_CMD, data, 2);
    if (ret < 0)
        return ret;

    msleep(AHT30_INIT_DELAY);
    return 0;
}


/*
 * Read sensor status (1 byte)
 */
static int aht30_read_status(struct i2c_client *client, u8 *status)
{
    return aht30_read_data(client, status, 1);
}

/* Removed - functionality merged into aht30_read_measurement */


/*
 * Read temperature and humidity measurement data from the sensor
 * Store results in struct aht30_data
 */
static int aht30_read_measurement(struct aht30_data *aht30)
{
    struct i2c_client *client = aht30->client;
    u8 cmd[3] = {0xAC, 0x33, 0x00}; // Measurement command
    u8 buf[6];
    u32 hum_raw, temp_raw;
    int ret;

    if (!client)
        return -ENODEV;

    // Send measurement command
    ret = i2c_master_send(client, cmd, 3);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to send measurement command: %d\n", ret);
        return ret;
    }

    // Wait for sensor to complete measurement
    msleep(80);

    // Read measurement data
    ret = i2c_master_recv(client, buf, 6);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to read measurement data: %d\n", ret);
        return ret;
    }

    // Check if sensor is busy
    if (buf[0] & AHT30_STATUS_BUSY) {
        dev_warn(&client->dev, "Sensor is busy\n");
        return -EBUSY;
    }

    // Decode humidity data (20 bits)
    hum_raw = ((buf[1] << 16) | (buf[2] << 8) | buf[3]) >> 4;
    aht30->humidity = (hum_raw * 1000) / 1048576; /* Convert to millipercent */

    // Decode temperature data (20 bits)
    temp_raw = ((buf[3] & 0x0F) << 16) | (buf[4] << 8) | buf[5];
    aht30->temperature = ((temp_raw * 2000) / 1048576) - 500; /* Convert to millidegree Celsius */

    return 0;
}


/*
 * Update sensor data if timeout has passed or data is invalid
 * Returns pointer to struct aht30_data
 */
static struct aht30_data *aht30_update_device(struct device *dev)
{
    struct aht30_data *aht30 = dev_get_drvdata(dev);
    int ret;

    mutex_lock(&aht30->lock);

    // If more than 1 second has passed or data is invalid, read again
    if (time_after(jiffies, aht30->last_updated + HZ) || !aht30->valid) {
        ret = aht30_read_measurement(aht30);
        if (ret < 0) {
            dev_err(dev, "Failed to read measurement: %d\n", ret);
            mutex_unlock(&aht30->lock);
            return ERR_PTR(ret);
        }
        aht30->last_updated = jiffies;
        aht30->valid = true;
    }

    mutex_unlock(&aht30->lock);
    return aht30;
}

/* Sysfs attributes */

/*
 * Export temperature data to sysfs (temp1_input)
 */
static ssize_t temp1_input_show(struct device *dev,
                               struct device_attribute *attr, char *buf)
{
    struct aht30_data *aht30 = aht30_update_device(dev);

    if (IS_ERR(aht30))
        return PTR_ERR(aht30);

    return sprintf(buf, "%d\n", aht30->temperature);
}


/*
 * Export humidity data to sysfs (humidity1_input)
 */
static ssize_t humidity1_input_show(struct device *dev,
                                   struct device_attribute *attr, char *buf)
{
    struct aht30_data *aht30 = aht30_update_device(dev);

    if (IS_ERR(aht30))
        return PTR_ERR(aht30);

    return sprintf(buf, "%d\n", aht30->humidity);
}

static DEVICE_ATTR_RO(temp1_input);
static DEVICE_ATTR_RO(humidity1_input);


// List of sysfs attributes exported to user
static struct attribute *aht30_attrs[] = {
    &dev_attr_temp1_input.attr,
    &dev_attr_humidity1_input.attr,
    NULL
};


// Create sysfs attribute group for hwmon
ATTRIBUTE_GROUPS(aht30);

/* I2C driver functions */

/*
 * Probe function: called when kernel detects compatible I2C device
 * Performs initialization, reset, calibration check, and hwmon device registration
 */
static int aht30_probe(struct i2c_client *client, const struct i2c_device_id *id)
{
    struct aht30_data *aht30;
    struct device *hwmon_dev;
    u8 status;
    int ret;

    dev_info(&client->dev, "AHT30 sensor probing\n");

    // Check if adapter supports I2C
    if (!i2c_check_functionality(client->adapter, I2C_FUNC_I2C)) {
        dev_err(&client->dev, "I2C adapter doesn't support I2C_FUNC_I2C\n");
        return -ENODEV;
    }

    // Allocate memory for struct aht30_data
    aht30 = devm_kzalloc(&client->dev, sizeof(*aht30), GFP_KERNEL);
    if (!aht30)
        return -ENOMEM;

    aht30->client = client;
    mutex_init(&aht30->lock);

    // Reset sensor
    ret = aht30_reset(client);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to reset sensor: %d\n", ret);
        return ret;
    }

    // Initialize sensor
    ret = aht30_init(client);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to initialize sensor: %d\n", ret);
        return ret;
    }

    // Check calibration status
    ret = aht30_read_status(client, &status);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to read status: %d\n", ret);
        return ret;
    }

    if (!(status & AHT30_STATUS_CALIBRATED)) {
        dev_err(&client->dev, "Sensor not calibrated\n");
        return -ENODEV;
    }

    dev_info(&client->dev, "AHT30 sensor initialized successfully\n");

    // Register hwmon device to export data to sysfs
    hwmon_dev = devm_hwmon_device_register_with_groups(&client->dev,
                                                       client->name,
                                                       aht30,
                                                       aht30_groups);
    if (IS_ERR(hwmon_dev)) {
        dev_err(&client->dev, "Failed to register hwmon device\n");
        return PTR_ERR(hwmon_dev);
    }

    i2c_set_clientdata(client, aht30);

    return 0;
}


/*
 * Remove function: called when driver is removed
 */
static int aht30_remove(struct i2c_client *client)
{
    dev_info(&client->dev, "AHT30 sensor removed\n");
    return 0;
}


// I2C device ID table for driver
static const struct i2c_device_id aht30_id[] = {
    { "aht30", 0 },
    { }
};
MODULE_DEVICE_TABLE(i2c, aht30_id);

// Device Tree match table
static const struct of_device_id aht30_of_match[] = {
    { .compatible = "aosong,aht30" },
    { }
};
MODULE_DEVICE_TABLE(of, aht30_of_match);


// Define I2C driver for AHT30 sensor
static struct i2c_driver aht30_driver = {
    .driver = {
        .name = "aht30",
        .of_match_table = aht30_of_match,
    },
    .probe = aht30_probe,
    .remove = aht30_remove,
    .id_table = aht30_id,
};

// Macro to register driver with kernel
module_i2c_driver(aht30_driver);

MODULE_AUTHOR("Nguyen Chi Tam");
MODULE_DESCRIPTION("AHT30 Temperature and Humidity Sensor Driver");
MODULE_LICENSE("GPL v2");
```
---

## 2. Update Kconfig

1. Edit the `Kconfig` file in the same directory.
2. Add the following entry:
    ```text
    config SENSORS_AHT30
        tristate "AHT30 humidity and temperature sensor"
        depends on I2C
        help
          Say Y here to enable support for AHT30 sensor.
    ```

---

## 3. Update Makefile

1. Edit the `Makefile` in the same directory.
2. Add the following line to build the new driver:
    ```text
    obj-$(CONFIG_SENSORS_AHT30) += aht30.o
    ```

---

## 4. Device Tree Configuration

1. Find the device tree source file for your board, typically located in:
    ```
    Build_Luckfox/luckfox-pico/sysdrv/source/kernel/arch/arm/boot/dts/rv1106-luckfox-pico-ultra-w.dts
    ```
- To ensure the driver works properly, you must add the following pinctrl configuration and AHT30 node to your board's device tree (.dts) file:
    ```dts
    &pinctrl {

        /* Existing content of rv1106-luckfox-pico-ultra-w.dts */

        pcfg_pull_none_smt: pcfg-pull-none-smt {
            u-boot,dm-spl;
        };

        i2c2 {
            i2c2m0_xfer: i2c2m0-xfer {
                u-boot,dm-spl;
            };
        };
    };

    &i2c2 {
        pinctrl-names = "default";
        pinctrl-0 = <&i2c2m0_xfer>;
        status = "okay";
        u-boot,dm-spl;                /* keep the controller */

        aht30@38 {
            compatible = "aosong,aht30";
            reg = <0x38>;
        };
    };
    ```

- Make sure these nodes are placed in the correct location in your board's .dts file.
- If this configuration is missing, the kernel will not properly initialize I2C2 and the AHT30 driver will not work.

---

## 5. Configure Kernel with Menuconfig

1. Go to the SDK root directory:
    ```bash
    cd Build_Luckfox/luckfox-pico/
    ```
2. Run kernel configuration:
    ```bash
    ./build.sh kernelconfig
    ```
3. In the menuconfig interface, navigate:
    ```
    Device Drivers  --->
        Industrial I/O support  --->
            [*] Humidity sensors  --->
                <*> AHT30 humidity and temperature sensor
    ```
4. Enable the AHT30 sensor option and save the configuration.

---

## 6. Save and Move Kernel Config

1. After saving, generate the defconfig file and move it to the configs directory:
    ```bash
    make ARCH=arm savedefconfig                     (generate defconfig file)
    mv defconfig arch/arm/configs/luckfox_rv1106_linux_custom_defconfig  (move new created file back to /configs folder)
    ```

---

## 7. Build and Flash the Kernel

1. Build the kernel and flash it to your board following the SDK instructions.

---

## 8. Verify Driver Functionality

1. Detect the sensor on I2C bus:
    ```bash
    i2cdetect -y 2
    ```
2. Check sensor readings:
    ```bash
    cat /sys/class/hwmon/hwmon1/temp1_input
    cat /sys/class/hwmon/hwmon1/humidity1_input
    ```

---

## 9. Remove Driver (Optional)

1. List I2C devices:
    ```bash
    ls /sys/bus/i2c/devices/
    ```
2. Unbind the driver:
    ```bash
    DEV="2-0038"
    DRV="$(readlink /sys/bus/i2c/devices/$DEV/driver | xargs basename)"
    echo "$DEV" | tee "/sys/bus/i2c/drivers/$DRV/unbind"
    ```

---

**You have now successfully added the AHT30 sensor package to the kernel menuconfig for the Luckfox Pico Ultra W board.**
